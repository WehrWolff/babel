program             : statement_list

terminator          : SEMICOLON

statement_list      : statement
                    | statement statement_list

statement           : single_stmt_list
                    | compound_stmt terminator

single_stmt_list    : single_stmt terminator
                    | single_stmt terminator single_stmt_list

single_stmt         : continue_stmt
                    | break_stmt
                    | goto_stmt
                    | label_stmt
                    | return_stmt
                    | raise_stmt
                    | import_stmt
                    | simple_stmt

simple_stmt         : NOOP
                    | assignment
                    | short_declaration
                    | element_assignment
                    | indirect_assignment
                    | expression

short_declaration   : VAR COLON_EQUALS expression

continue_stmt       : CONTINUE
                    | CONTINUE VAR

break_stmt          : BREAK
                    | BREAK VAR

goto_stmt           : GOTO VAR

label_stmt          : LABEL_START VAR

return_stmt         : RETURN comma_values
                    | RETURN

raise_stmt          : RAISE expression

import_stmt         : IMPORT comma_values

element_assignment  : VAR LSQUARE expression RSQUARE assignment_operator expression

chained_indirection : MULTIPLY
                    | MULTIPLY chained_indirection

indirect_assignment : VAR chained_indirection assignment_operator expression

assignment          : VAR assignment_operator expression
                    | LET VAR assignment_operator expression
                    | CONST VAR assignment_operator expression
                    | LET VAR type_spec assignment_operator expression
                    | CONST VAR type_spec assignment_operator expression

assignment_operator : EQUALS
                    | PLUS_EQUALS
                    | MINUS_EQUALS
                    | MULTIPLY_EQUALS
                    | DIVIDE_EQUALS
                    | POWER_EQUALS
                    | MODULO_EQUALS
                    | INTEGER_DIVIDE_EQUALS
                    | LSHIFT_EQUALS
                    | RSHIFT_EQUALS
                    | BIT_OR_EQUALS
                    | BIT_AND_EQUALS
                    | BIT_XOR_EQUALS

type_spec           : COLON type

type                : TYPE
                    | VAR LT generic_list GT
                    | MULTIPLY CONST type
                    | MULTIPLY type

generic             : type
                    | atom

generic_list        : generic 
                    | generic COMMA generic_list

compound_stmt       : task_def
                    | class_def
                    | struct_def
                    | control_flow
                    | loop_stmt
                    | try_stmt

block               : statement_list
                    | single_stmt_list

args                : VAR type_spec
                    | VAR type_spec EQUALS expression
                    | VARARG
                    | args COMMA args

type_signature      : type
                    | VARARG
                    | type COMMA type_signature

task_header         : TASK VAR LPAREN RPAREN RARR type DO
                    | TASK VAR LPAREN args RPAREN RARR type DO

extern_task         : EXTERN TASK VAR LPAREN type_signature RPAREN RARR type
                    | EXTERN TASK VAR LPAREN RPAREN RARR type

task_def            : task_header block END
                    | extern_task

task_def_list       : task_def
                    | task_def task_def_list

members             : VAR terminator
                    | VAR type_spec terminator
                    | VAR terminator members
                    | VAR type_spec terminator members

struct_def          : STRUCT VAR NEWLINE members END

class_def           : CLASS VAR NEWLINE members task_def_list END

control_flow        : if_stmt
                    | match_stmt

if_stmt             : IF expression THEN block END
                    | IF expression THEN block elif_stmt
                    | IF expression THEN block ELSE block END

elif_stmt           : ELIF expression THEN block END
                    | ELIF expression THEN block elif_stmt
                    | ELIF expression THEN block ELSE block END

match_stmt          : MATCH expression NEWLINE cases
                    | MATCH expression NEWLINE cases OTHERWISE block

cases               : CASE expression block
                    | CASE expression block cases

loop_stmt           : for_loop
                    | for_in_loop
                    | while_loop

for_loop            : FOR simple_stmt SEMICOLON expression SEMICOLON simple_stmt DO block END
                    | LOOP_LABEL_START VAR COLON FOR simple_stmt SEMICOLON expression SEMICOLON simple_stmt DO block END

for_in_loop         : FOR VAR IN VAR DO block END
                    | LOOP_LABEL_START VAR COLON FOR VAR IN VAR DO block END

while_loop          : WHILE expression DO block END
                    | LOOP_LABEL_START VAR COLON WHILE expression DO block END

try_stmt            : TRY block catch_blocks END
                    | TRY block catch_blocks FINALLY block END

catch_blocks        : CATCH VAR block
                    | CATCH VAR block catch_blocks

expression          : disjunction

disjunction         : contravalence
                    | contravalence OR disjunction

contravalence       : conjunction
                    | conjunction XOR contravalence

conjunction         : inversion
                    | inversion AND conjunction

inversion           : NOT inversion
                    | comparison

comparison          : comparison EQEQ bitwise_or
                    | comparison NOTEQ bitwise_or
                    | comparison LTEQ bitwise_or
                    | comparison GTEQ bitwise_or
                    | comparison LT bitwise_or
                    | comparison GT bitwise_or
                    | bitwise_or

bitwise_or          : bitwise_or BIT_OR bitwise_xor
                    | bitwise_xor

bitwise_xor         : bitwise_xor BIT_XOR bitwise_and
                    | bitwise_and

bitwise_and         : bitwise_and BIT_AND shift_expression
                    | shift_expression

shift_expression    : shift_expression LSHIFT sum
                    | shift_expression RSHIFT sum
                    | sum

sum                 : sum PLUS term
                    | sum MINUS term
                    | term

term                : term MULTIPLY exponentiation
                    | term DIVIDE exponentiation
                    | term INTEGER_DIVIDE exponentiation
                    | term MODULO exponentiation
                    | exponentiation

exponentiation      : prefix POWER exponentiation
                    | prefix

prefix              : PLUS prefix
                    | MINUS prefix
                    | INCREMENT prefix
                    | DECREMENT prefix
                    | BIT_NOT prefix
                    | BIT_AND prefix
                    | postfix

postfix             : postfix DOT VAR
                    | postfix DOT VAR LPAREN RPAREN
                    | postfix DOT VAR LPAREN params RPAREN
                    | postfix LSQUARE expression RSQUARE
                    | postfix MULTIPLY
                    | postfix INCREMENT
                    | postfix DECREMENT
                    | primary

primary             : atom
                    | macro_call
                    | function_call
                    | class_construction
                    | LPAREN expression RPAREN

macro_call          : AT VAR
                    | AT VAR LPAREN generic_list RPAREN

function_call       : VAR LPAREN RPAREN
                    | VAR LPAREN params RPAREN

class_construction  : NEW VAR LPAREN RPAREN
                    | NEW VAR LPAREN params RPAREN

param               : VAR EQUALS expression
                    | expression

params              : param COMMA params
                    | param

atom                : STRING
                    | CSTRING
                    | CHAR
                    | BOOL
                    | INTEGER
                    | FLOATING_POINT
                    | NULL
                    | VAR
                    | list
                    | tuple
                    | set
                    | map

comma_values        : expression
                    | expression COMMA comma_values

list                : LSQUARE comma_values RSQUARE
                    | LSQUARE RSQUARE

tuple               : LPAREN comma_values RPAREN
                    | LPAREN RPAREN

set                 : LBRACE comma_values RBRACE
                    | LBRACE RBRACE

map                 : LBRACE kvpairs RBRACE
                    | LBRACE RBRACE
